def CDP(close, high, low):
    cdp_list = []
    ah_list = []
    nh_list = []
    al_list = []
    nl_list = []
    for i in range(len(close)):
        cdp = (high[i]+low[i] + 2 *close[i])/4
        ah = cdp + high[i] - low[i]
        nh = cdp * 2 - low[i]
        al = cdp - (high[i] - low[i])
        nl = cdp * 2 - high[i]

        cdp_list = round(cdp,2)
        ah_list = round(ah,2)
        nh_list = round(nh,2)
        al_list = round(al,2)
        nl_list = round(nl,2)

    return ah_list, nh_list, cdp_list, nl_list, al_list
